__Update Data by WHERE clause_

NOTE:: Be careful when updating records in a table;

UPDATE Syntax

UPDATE tablename
SET column1=value1, column2=value2,...
WHERE condition;

eg Syntax

-UPDATE customers
-SET contactname = 'maung maung', city='mandalay'
-WHERE customerid =1;
